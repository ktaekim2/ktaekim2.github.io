Scanner scan = new Scanner(System.in);
int num[] = { 5, 2, 1, 3, 4, 7, 6, 9, 10, 8 };
int input = 0;

while (true) {
  System.out.print("숫자: ");
  input = scan.nextInt();
  for (int i = 0; i < num.length; i++) {
    if (num[i] == input) {
      System.out.println(input + "는 " + i + "번 인덱스에 있습니다.");
    }

  }
}
